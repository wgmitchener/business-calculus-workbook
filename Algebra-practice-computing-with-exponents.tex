Express each of these as an integer or a fraction in lowest terms.

\begin{multicols}{2}

 \begin{ProblemSet}[pencil space=1in]
  \EqProb{-2^{-1}}
  \EqProb{(-2)^{-1}}

  \EqProb{0^2}
  \EqProb{0^1}
  \EqProb{0^0}
  \EqProb{0^{-1}}

  \EqProb{1^2}
  \EqProb{1^1}
  \EqProb{1^{-1}}
  \EqProb{-1^2}
  \EqProb{(-1)^2}
  \EqProb{(-1)^3}

  \EqProb{-2^2}
  \EqProb{-(2^2)}
  \EqProb{(-2)^2}
  \EqProb{-2^3}
  \EqProb{-(2^3)}
  \EqProb{(-2)^3}
  \EqProb{-2^{-2}}
  \EqProb{-(2^{-2})}
  \EqProb{(-2)^{-2}}
  \EqProb{-2^{-3}}
  \EqProb{-(2^{-3})}
  \EqProb{(-2)^{-3}}

  \EqProb{\pfrac{1}{2}^{-1}}
  \EqProb{-\pfrac{1}{2}^{-1}}
  \EqProb{\left(-\pfrac{1}{2}\right)^{-1}}
 \end{ProblemSet}
\end{multicols}
