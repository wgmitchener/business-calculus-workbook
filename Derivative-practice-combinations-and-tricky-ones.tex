\Section{Derivative practice---combinations and tricky ones}

\begin{ProblemSet}[pencil space=3in]

 \begin{Problem}
  What is $f'(x)$?
  \begin{equation*}
   f(x) = x^2 \left(x^2 + 3x\right)^{-2}
  \end{equation*}
 \end{Problem}

 \begin{Problem}
  What is $\Deriv{y}{t}$?
  \begin{equation*}
   y = \left(\frac{4t + 3}{t^2 - 1}\right)^{\nicefrac{3}{5}}
  \end{equation*}
 \end{Problem}

 \begin{Problem}
  What is $h'(z)$?
  \begin{equation*}
   h(z) = \frac{\left(z^5 - 4z\right)^{\nicefrac{1}{5}}}{5z - 3}
  \end{equation*}
 \end{Problem}

 \begin{Problem}
  What is $g'(t)$?
  \begin{equation*}
   g(t) = t\cdot(t+1)\cdot\left(3t^2 + 2\right)
  \end{equation*}
 \end{Problem}

\end{ProblemSet}
